ARGUMENT X,Y,Z;

VARIABLE f=X;
VARIABLE i=Y;
VARIABLE k=Z;

//Шестая подсистема

CONSTANT a6=1, b6=0.8;
VARIABLE W61=a6*f^b6;
 
CONSTANT d6=1, c6=0.2;
VARIABLE W62=d6*i^c6;
 
CONSTANT e6=1, g6=0;
VARIABLE W63=e6*k+g6;

CONSTANT m6=1, l6=0.2;
VARIABLE W64=m6*i^l6;
 
VARIABLE W65=W61&W62&W63&W64;
 
RETURN W65;
