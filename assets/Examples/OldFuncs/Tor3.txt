ARGUMENT X,Y,Z;

//x : 6 ; y : 6 ; z : 6

//CONSTANT PI=3.141593;
CONSTANT PI=3.1415926536;//5897932384626433832795;
CONSTANT R=5;

VARIABLE AL1=R*ARCTAN(Y/X);

VARIABLE AL=((1+ABS(X)/X)/2)*(PI/2 )+AL1/PI;
VARIABLE X1=SQRT(X*X+Y*Y)-R;
VARIABLE XX=X1*COS(AL)+Z*SIN(AL);
VARIABLE ZZ=-X1*SIN(AL)+Z*COS(AL);
VARIABLE F1=1-XX*XX-ZZ*ZZ/4;

RETURN F1;
