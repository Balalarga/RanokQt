ARGUMENT X,Y,Z;

VARIABLE f=X;
VARIABLE i=Y;
VARIABLE k=Z;


//Пятая подсистема

CONSTANT a5=2, b5=0.4;
VARIABLE W51=a5*f^b5;
 
CONSTANT d5=3, c5=0.7;
VARIABLE W52=d5*i^c5;
 
CONSTANT e5=2, g5=1;
VARIABLE W53=e5*k+g5;

CONSTANT m5=3, l5=0.2;
VARIABLE W54=m5*i^l5;
 
VARIABLE W55=W51&W52&W53&W54;

RETURN W55;
